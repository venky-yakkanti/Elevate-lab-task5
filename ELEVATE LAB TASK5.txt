CREATE TABLE CUSTOMERS(
CUSTOMER_ID INT PRIMARY KEY,
NAME VARCHAR(20)
);

CREATE TABLE ORDERS(
ORDER_ID INT PRIMARY KEY,
CUSTIMER_ID INT,
PRODUCT VARCHAR(10),
FOREIGN KEY (CUSTOMER_ID) REFERENCE CUSTOMERS(CUSTOMER_ID)
);
